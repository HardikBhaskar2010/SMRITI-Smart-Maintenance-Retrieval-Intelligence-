```typescript
interface AssetState {
  // Cached asset list (persisted for offline use)
  assets: AssetSummary[];
  setAssets: (assets: AssetSummary[]) => void;
  updateAssetDebt: (assetId: string, score: number, severity: string) => void;

  // Last fetch time (for stale-while-revalidate logic)
  lastFetched: number | null;
  setLastFetched: (ts: number) => void;

  // Derived: summary counts
  getCriticalCount: () => number;
  getWarningCount: () => number;
  getOkCount: () => number;
}

export const useAssetStore = create<AssetState>()(
  persist(
    (set, get) => ({
      assets: [],
      setAssets: (assets) => set({ assets }),
      updateAssetDebt: (assetId, score, severity) =>
        set((s) => ({
          assets: s.assets.map((a) =>
            a.asset_id === assetId
              ? { ...a, debt_score: score, severity: severity as any }
              : a
          ),
        })),

      lastFetched: null,
      setLastFetched: (ts) => set({ lastFetched: ts }),

      getCriticalCount: () =>
        get().assets.filter((a) => a.severity === "CRITICAL").length,
      getWarningCount: () =>
        get().assets.filter((a) => a.severity === "WARNING").length,
      getOkCount: () =>
        get().assets.filter((a) => a.severity === "OK").length,
    }),
    {
      name: "smriti-assets",
      // Only persist the asset list, not derived fn references
      partialize: (s) => ({ assets: s.assets, lastFetched: s.lastFetched }),
    }
  )
);
```

### `src/hooks/useVoice.ts` — Web Speech API

```typescript
import { useCallback, useEffect, useRef, useState } from "react";

export type VoiceStatus = "idle" | "listening" | "processing" | "error" | "unsupported";

interface UseVoiceOptions {
  language?: string;               // default: "hi-IN" (Hindi with English fallback)
  silenceTimeout?: number;         // ms of silence before auto-submit (default: 1500)
  onTranscript?: (text: string) => void;
  onFinalTranscript?: (text: string) => void;
  onError?: (error: string) => void;
}

interface UseVoiceReturn {
  status: VoiceStatus;
  transcript: string;              // Live, rolling transcript
  startListening: () => void;
  stopListening: () => void;
  isListening: boolean;
  isSupported: boolean;
}

export function useVoice({
  language = "hi-IN",
  silenceTimeout = 1500,
  onTranscript,
  onFinalTranscript,
  onError,
}: UseVoiceOptions = {}): UseVoiceReturn {
  const [status, setStatus] = useState<VoiceStatus>("idle");
  const [transcript, setTranscript] = useState("");
  const recognitionRef = useRef<SpeechRecognition | null>(null);
  const silenceTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  // Check browser support
  const isSupported =
    typeof window !== "undefined" &&
    ("SpeechRecognition" in window || "webkitSpeechRecognition" in window);

  useEffect(() => {
    if (!isSupported) {
      setStatus("unsupported");
      return;
    }

    const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
    const recognition = new SR();

    recognition.continuous = true;
    recognition.interimResults = true;
    recognition.lang = language;
    recognition.maxAlternatives = 1;

    recognition.onstart = () => {
      setStatus("listening");
      setTranscript("");
    };

    recognition.onresult = (event: SpeechRecognitionEvent) => {
      let interim = "";
      let finalText = "";

      for (let i = event.resultIndex; i < event.results.length; i++) {
        const result = event.results[i];
        if (result.isFinal) {
          finalText += result[0].transcript;
        } else {
          interim += result[0].transcript;
        }
      }

      const current = finalText || interim;
      setTranscript(current);
      onTranscript?.(current);

      // Reset silence timer on new speech
      if (silenceTimerRef.current) clearTimeout(silenceTimerRef.current);
      if (finalText) {
        silenceTimerRef.current = setTimeout(() => {
          recognition.stop();
          onFinalTranscript?.(finalText);
        }, silenceTimeout);
      }
    };

    recognition.onerror = (event: SpeechRecognitionErrorEvent) => {
      const msg = event.error === "not-allowed"
        ? "Microphone permission denied. Please allow access and try again."
        : `Voice recognition error: ${event.error}`;
      setStatus("error");
      onError?.(msg);
    };

    recognition.onend = () => {
      if (silenceTimerRef.current) clearTimeout(silenceTimerRef.current);
      setStatus("idle");
    };

    recognitionRef.current = recognition;

    return () => {
      recognition.abort();
      if (silenceTimerRef.current) clearTimeout(silenceTimerRef.current);
    };
  }, [isSupported, language, silenceTimeout]);

  const startListening = useCallback(() => {
    if (!recognitionRef.current || status === "listening") return;
    setTranscript("");
    try {
      recognitionRef.current.start();
    } catch {
      // Already started — ignore
    }
  }, [status]);

  const stopListening = useCallback(() => {
    recognitionRef.current?.stop();
  }, []);

  return {
    status,
    transcript,
    startListening,
    stopListening,
    isListening: status === "listening",
    isSupported,
  };
}
```

### `src/hooks/useIngestion.ts` — WebSocket Progress

```typescript
import { useCallback, useRef, useState } from "react";
import type { IngestProgress, IngestResult } from "@/api/types";

const WS_URL = import.meta.env.VITE_WS_URL ?? "ws://localhost:8000";

interface UseIngestionReturn {
  progress: IngestProgress | null;
  result: IngestResult | null;
  isUploading: boolean;
  error: string | null;
  uploadFile: (file: File) => Promise<void>;
  reset: () => void;
}

export function useIngestion(): UseIngestionReturn {
  const [progress, setProgress] = useState<IngestProgress | null>(null);
  const [result, setResult] = useState<IngestResult | null>(null);
  const [isUploading, setIsUploading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const wsRef = useRef<WebSocket | null>(null);

  const uploadFile = useCallback(async (file: File) => {
    setIsUploading(true);
    setError(null);
    setResult(null);
    setProgress(null);

    return new Promise<void>((resolve, reject) => {
      const ws = new WebSocket(`${WS_URL}/ws/ingest`);
      wsRef.current = ws;

      ws.onopen = async () => {
        // Convert file to base64 and send
        const buffer = await file.arrayBuffer();
        const bytes = new Uint8Array(buffer);
        let binary = "";
        bytes.forEach((b) => (binary += String.fromCharCode(b)));
        const b64 = btoa(binary);

        ws.send(
          JSON.stringify({
            filename: file.name,
            data: b64,
          })
        );
      };

      ws.onmessage = (event) => {
        const data = JSON.parse(event.data);
        if (data.type === "result") {
          setResult(data as IngestResult);
          setIsUploading(false);
          ws.close();
          resolve();
        } else if (data.type === "error") {
          setError(data.message);
          setIsUploading(false);
          ws.close();
          reject(new Error(data.message));
        } else {
          // Progress update
          setProgress(data as IngestProgress);
        }
      };

      ws.onerror = () => {
        const msg = "WebSocket connection failed. Is the backend running?";
        setError(msg);
        setIsUploading(false);
        reject(new Error(msg));
      };
    });
  }, []);

  const reset = useCallback(() => {
    wsRef.current?.close();
    setProgress(null);
    setResult(null);
    setIsUploading(false);
```

```tsx
// Force-directed layout: pre-compute positions using a simple spring algorithm
function computeLayout(nodes: GraphNode[]): Record<string, [number, number, number]> {
  const positions: Record<string, [number, number, number]> = {};
  const count = nodes.length;

  nodes.forEach((node, i) => {
    // Spherical distribution
    const phi   = Math.acos(-1 + (2 * i) / count);
    const theta = Math.sqrt(count * Math.PI) * phi;
    const r     = 6 + (node.item_count / 30);        // Larger thread = further from center

    positions[node.id] = [
      r * Math.sin(phi) * Math.cos(theta),
      r * Math.sin(phi) * Math.sin(theta),
      r * Math.cos(phi),
    ];
  });

  return positions;
}

export function KnowledgeGraph() {
  const { data: graphData, isLoading } = useGraphData();
  const { hoveredNodeId, setHoveredNode, openAssetDrawer } = useUIStore();

  const positions = useMemo(
    () => (graphData ? computeLayout(graphData.nodes) : {}),
    [graphData]
  );

    <div className="graph-container" style={{ width: "100%", height: "100%" }}>
```

```tsx
interface AssetNodeProps {
  node: GraphNode;
  position: [number, number, number];
  isHovered: boolean;
  onPointerOver: () => void;
  onPointerOut: () => void;
  onClick: () => void;

// Map severity → emissive color
const SEVERITY_COLORS = {
  OK:       new THREE.Color("#22C55E"),
  WARNING:  new THREE.Color("#F59E0B"),
  CRITICAL: new THREE.Color("#EF4444"),
};

export function AssetNode({
  node, position, isHovered,
  onPointerOver, onPointerOut, onClick,
}: AssetNodeProps) {
  const meshRef = useRef<THREE.Mesh>(null);
  const pulseRef = useRef(0);
  const baseColor = SEVERITY_COLORS[node.severity];

  // Node radius: proportional to thread item count, clamped 0.3–1.2
  const radius = Math.min(1.2, Math.max(0.3, 0.3 + node.item_count / 100));

  useFrame((_, delta) => {
    if (!meshRef.current) return;

    // CRITICAL: pulsing glow effect
    if (node.severity === "CRITICAL") {
      pulseRef.current += delta * 2;
      const pulse = 0.5 + 0.5 * Math.sin(pulseRef.current);
      (meshRef.current.material as THREE.MeshStandardMaterial).emissiveIntensity =
        0.4 + pulse * 0.6;
    }

    // Hover: scale up
    const targetScale = isHovered ? 1.3 : 1.0;
    meshRef.current.scale.lerp(
      new THREE.Vector3(targetScale, targetScale, targetScale),
      0.15
    );
  });

  return (
    <group position={position}>
      <Sphere
        ref={meshRef}
        args={[radius, 32, 32]}
        onClick={onClick}
        onPointerOver={onPointerOver}
        onPointerOut={onPointerOut}
      >
        <meshStandardMaterial
          color={baseColor}
          emissive={baseColor}
          emissiveIntensity={node.severity === "CRITICAL" ? 0.6 : 0.2}
          roughness={0.3}
          metalness={0.4}
        />
      </Sphere>
```
